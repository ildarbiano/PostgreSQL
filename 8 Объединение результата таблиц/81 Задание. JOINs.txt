FROM sklad

INNER JOIN sales

  ON sklad.name = sales.name;
  

-- left

SELECT sklad.name, sklad.price, sales.quant

FROM sklad

LEFT JOIN sales

  ON sklad.name = sales.name;

FROM sklad
